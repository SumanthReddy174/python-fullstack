def prime_numbers(n):
    for num in range(2, n + 1):  # check numbers from 2 to n
        for i in range(2, int(num ** 0.5) + 1):  # check divisors up to sqrt(num)
            if num % i == 0:  # if divisible, not prime
                break
        else:
            yield num  # if no break, it's a prime â†’ yield it
