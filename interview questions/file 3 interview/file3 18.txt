18. Implement a recursive function to compute Fibonacci numbers.

def fibonacci(n):
    if n == 0:
        return 0   # base case
    elif n == 1:
        return 1   # base case
    else:
        return fibonacci(n-1) + fibonacci(n-2)  # recursive call

# Test: Print first 10 Fibonacci numbers
for i in range(10):
    print(fibonacci(i), end=" ")
